CREATE PROC [dbo].[branch_add_branch]
@branch_name VARCHAR(50),
@branch_location VARCHAR(50),
@branch_code VARCHAR(50) = NULL,
@branch_phone_number VARCHAR(50),
@branch_email VARCHAR(50)

AS

DECLARE @recordsFirtstCount INT;
DECLARE @recordsSecondCount INT;

DECLARE @result INT = 0;

BEGIN

SET @recordsFirtstCount = (SELECT COUNT (branchID) FROM tbl_branches);

INSERT INTO tbl_branches
(
branch_name,
branch_location,
branch_code,
branch_phone_number,
branch_email,
status
)

VALUES
(
@branch_name,
@branch_location,
@branch_code,
@branch_phone_number,
@branch_email,
'true'
)

SET @recordsSecondCount = (SELECT COUNT (branchID) FROM tbl_branches);


IF (@recordsFirtstCount + 1 ) = @recordsSecondCount
SET @result = 1;
ELSE
SET @result = 0;

SELECT @result AS result
END
